--- script.js (jQuery version) // --- Cookies Helper Functions --- function setCookie(name, value, minutes) { const d = new Date(); d.setTime(d.getTime() + (minutes * 60 * 1000)); document.cookie = `${name}=${value};expires=${d.toUTCString()};path=/`; } function getCookie(name) { const cname = name + "="; const decodedCookie = decodeURIComponent(document.cookie); const ca = decodedCookie.split(';'); for (let c of ca) { c = c.trim(); if (c.indexOf(cname) === 0) { return c.substring(cname.length, c.length); } } return ""; } $(document).ready(function () { // --- Page 1 (index.html) --- $("#loginForm").on("submit", function (e) { e.preventDefault(); let username = $("#username").val(); if (username.trim() !== "") { setCookie("username", username, 1); // 1 minute expiry window.location.href = "form.html"; } }); // --- Page 2 (form.html) --- if ($("#userForm").length) { let user = getCookie("username"); if (!user) { alert("Session expired! Please login again."); window.location.href = "index.html"; } // Load Table Data function loadTable() { let users = JSON.parse(localStorage.getItem("users")) || []; $("#dataTableBody").empty(); $.each(users, function (index, u) { let row = ` ${u.name} ${u.email} ${u.phone} ${u.dob} ${u.gender} ${u.address} `; $("#dataTableBody").append(row); }); } // Form Submit $("#userForm").on("submit", function (e) { e.preventDefault(); let users = JSON.parse(localStorage.getItem("users")) || []; let newUser = { name: $("#name").val(), email: $("#email").val(), phone: $("#phone").val(), dob: $("#dob").val(), gender: $("#gender").val(), address: $("#address").val() }; users.push(newUser); localStorage.setItem("users", JSON.stringify(users)); $("#userForm")[0].reset(); loadTable(); }); // Delete Button $(document).on("click", ".deleteBtn", function () { let index = $(this).data("index"); let users = JSON.parse(localStorage.getItem("users")) || []; users.splice(index, 1); localStorage.setItem("users", JSON.stringify(users)); loadTable(); }); // Edit Button $(document).on("click", ".editBtn", function () { let index = $(this).data("index"); let users = JSON.parse(localStorage.getItem("users")) || []; let u = users[index]; $("#name").val(u.name); $("#email").val(u.email); $("#phone").val(u.phone); $("#dob").val(u.dob); $("#gender").val(u.gender); $("#address").val(u.address); users.splice(index, 1); // remove old localStorage.setItem("users", JSON.stringify(users)); loadTable(); }); loadTable(); } }); --- ✅ இப்போ எல்லாமே jQuery + Tailwind Form submit, table render, edit/delete → எல்லாம் jQuery event handlers Style Tailwindலே maintain பண்ணிருக்கேன் --- 👉 Cookie expiry நான் 1 minute fix பண்ணிருக்கேன். நீங்க வேணும்னா இதை 5 mins மாத்தி வைக்கலாமா?
